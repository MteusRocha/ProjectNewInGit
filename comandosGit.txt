Commands git


<Para apagar o repositorio atual>
rm -rf .git/


<Configurar um novo usuario>
git config --global user."email/name" "texte"
ou
<Mostra todas as configuracoes do git>
git config --list


<Mostra os Commits executados>
git reflog


<Para voltar a um ponto de versao do git>
git reset --hard "mais o codigo do commit"



--BRANCHS

<Para listar as branchs disponiveis>
git branch

<criar uma nova branch>
git branch 'mais o nome da branch'


<Para mudar de Branch>
git checkout 'nome da branch'


--Atualizar e unir Branchs

<Para capturar as alteracoes feitas por outros colaboradores>
git pull

<Unir branchs>

git merge 'nome da branch que voce para unir'
